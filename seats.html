import { db, ref, get } from "./firebase.js"; document.addEventListener("DOMContentLoaded", async () => { const bubbles = document.getElementById('qty-bubbles'); const vehicleImg = document.getElementById('vehicle-img'); const popup = document.getElementById('seat-popup'); const matchTitle = document.getElementById('match-title'); const venueImg = document.getElementById('venue-img'); // Default selection let selectedSeats = 1; // ========================================== // 🚲 VEHICLE ICONS (Standard BMS Links) // ========================================== const vehicles = { 1: "https://in.bmscdn.com/webin/common/icons/bicycle.png", 2: "https://in.bmscdn.com/webin/common/icons/scooter.png", 3: "https://in.bmscdn.com/webin/common/icons/auto.png", 4: "https://in.bmscdn.com/webin/common/icons/mini-car.png", 5: "https://in.bmscdn.com/webin/common/icons/sedan-car.png", 6: "https://in.bmscdn.com/webin/common/icons/suv-car.png", 7: "https://in.bmscdn.com/webin/common/icons/suv-car.png", 8: "https://in.bmscdn.com/webin/common/icons/suv-car.png", 9: "https://in.bmscdn.com/webin/common/icons/suv-car.png", 10: "https://in.bmscdn.com/webin/common/icons/suv-car.png" }; // ========================================== // 1. 🔥 GENERATE BUBBLES FIRST (Taaki Numbers dikhein) // ========================================== function createBubbles() { if (!bubbles) return; bubbles.innerHTML = ""; for (let i = 1; i <= 10; i++) { const btn = document.createElement('div'); btn.className = 'qty-bubble'; if (i === 1) btn.classList.add('active'); btn.innerText = i; btn.addEventListener('click', () => { document.querySelectorAll('.qty-bubble').forEach(b => b.classList.remove('active')); btn.classList.add('active'); if (vehicleImg && vehicles[i]) { vehicleImg.src = vehicles[i]; } selectedSeats = i; localStorage.setItem("seatQty", selectedSeats); }); bubbles.appendChild(btn); } } // Pehle bubbles bana do taaki screen khali na dikhe createBubbles(); // ========================================== // 2. 🔥 LOAD DATA (LOCAL + FIREBASE SYNC) // ========================================== const rawData = localStorage.getItem('selectedMatch'); const matchId = localStorage.getItem("matchId"); if (rawData) { const match = JSON.parse(rawData); if (matchTitle) matchTitle.innerText = match.title || "Select Seats"; // Venue Image Setting if (venueImg) { venueImg.src = match.venue_img || match.banner || "https://via.placeholder.com/800x400?text=Loading+Layout..."; } } // Agar ID hai toh Firebase se fresh data uthao (Venue Image ke liye) if (matchId) { try { const snapshot = await get(ref(db, `matches/${matchId}`)); if (snapshot.exists()) { const data = snapshot.val(); if (matchTitle) matchTitle.innerText = data.title; if (venueImg) { // Priority: venue_img -> banner -> placeholder venueImg.src = data.venue_img || data.banner || ""; } } } catch (err) { console.error("Firebase Sync Error:", err); } } // ========================================== // 3. POPUP & CONFIRM LOGIC // ========================================== const confirmBtn = document.getElementById('confirm-btn'); if (confirmBtn) { confirmBtn.onclick = () => { if (popup) popup.classList.remove('active'); // Redirect to layout or next step }; } const openBtn = document.getElementById('open-seat-popup'); if (openBtn) { openBtn.onclick = () => { if (popup) popup.classList.add('active'); }; } });
